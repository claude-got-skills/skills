import { chat } from '@/app/actions/chat'

export function ChatWidget({ userDataContext }: { userDataContext: string }) {
  const [input, setInput] = useState('')
  const [messages, setMessages] = useState<{ role: string; content: string }[]>([])
  const [isLoading, setIsLoading] = useState(false)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!input.trim()) return

    const userMessage = input
    setInput('')
    setMessages((prev) => [...prev, { role: 'user', content: userMessage }])
    setIsLoading(true)

    try {
      const response = await chat(userMessage, userDataContext)
      const reader = response.getReader()
      const decoder = new TextDecoder()
      let fullMessage = ''

      // Add empty assistant message to update as we stream
      setMessages((prev) => [...prev, { role: 'assistant', content: '' }])

      while (true) {
        const { done, value } = await reader.read()
        if (done) break

        const chunk = decoder.decode(value)
        fullMessage += chunk

        // Update the last message with streamed content
        setMessages((prev) => [
          ...prev.slice(0, -1),
          { role: 'assistant', content: fullMessage },
        ])
      }
    } catch (error) {
      console.error('Chat error:', error)
      setMessages((prev) => [
        ...prev,
        { role: 'assistant', content: 'Error: Could not generate response' },
      ])
    } finally {
      setIsLoading(false)
    }
  }

  return (
    <div className="flex flex-col gap-4 w-full max-w-2xl">
      <div className="flex flex-col gap-2 h-96 overflow-y-auto border rounded p-4">
        {messages.map((msg, i) => (
          <div key={i} className={`p-2 rounded ${msg.role === 'user' ? 'bg-blue-100' : 'bg-gray-100'}`}>
            <strong>{msg.role}:</strong> {msg.content}
          </div>
        ))}
      </div>
      <form onSubmit={handleSubmit} className="flex gap-2">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Ask about your data..."
          disabled={isLoading}

          {isLoading ? 'Thinking...' : 'Send'}

      </form>
    </div>
  )
}
