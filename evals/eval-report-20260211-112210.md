```python
def detect_structure(blocks: list[dict]) -> dict:
    """Identify headings by font size, bold, position."""
    headings = []
    for b in blocks:
        if b.get("type") == 0:  # text block
            for line in b.get("lines", []):
                for span in line.get("spans", []):
                    if span["size"] > 14 or "Bold" in span.get("font", ""):

                            "text": span["text"],
                            "font_size": span["size"],
                            "position": b["bbox"],
                        })
```

100-page documents won't fit in a single LLM context (and even if they did, retrieval quality degrades). Chunk intelligently.

```python
from dataclasses import dataclass

@dataclass
class Chunk:
    chunk_id: str
    doc_id: str
    text: str
    page_range: tuple[int, int]    # (start_page, end_page)
    section_heading: str | None
    token_count: int

def chunk_document(
    doc: Document,
    max_tokens: int = 1000,
    overlap_tokens: int = 150,
) -> list[Chunk]:
    """
    Two-pass chunking:
      1. Try section-based splits (using detected headings)
      2. Fall back to sliding window if sections are too large
    """
    sections = split_by_sections(doc)
    chunks = []

    for section in sections:
        if section.token_count <= max_tokens:
            chunks.append(section)
        else:
            # Subdivide large sections with overlap
            chunks.extend(
                sliding_window_chunk(section, max_tokens, overlap_tokens)
            )

    return chunks

def split_by_sections(doc: Document) -> list[Chunk]:
    """Group pages by detected headings."""
    chunks = []
    current_text = ""
    current_heading = None
    start_page = 1

    for page in doc.pages:
        headings = page.metadata.get("headings", [])

        if headings and current_text.strip():
            # New heading found — flush the current chunk
            chunks.append(Chunk(
                chunk_id=f"{doc.doc_id}_s{len(chunks)}",
                doc_id=doc.doc_id,
                text=current_text.strip(),
                page_range=(start_page, page.page_num - 1),
                section_heading=current_heading,
                token_count=estimate_tokens(current_text),
            ))
            current_text = ""
            current_heading = headings[0]["text"]
            start_page = page.page_num

        current_text += f"\n{page.text}"

    # Don't forget the last chunk
    if current_text.strip():
        chunks.append(Chunk(
            chunk_id=f"{doc.doc_id}_s{len(chunks)}",
            doc_id=doc.doc_id,
            text=current_text.strip(),
            page_range=(start_page, doc.total_pages),
            section_heading=current_heading,
            token_count=estimate_tokens(current_text),
        ))

    return chunks
```

**Key decision:** Section-based chunking first, sliding-window fallback second. This preserves document structure, which matters a lot for summary quality.

    These are deterministic and fast — good first line of defense.

    def check(self, item: ContentItem) -> list[Issue]:
        import re
        issues = []

        for rule in self.rules:
            # Pattern-based rules
            if "pattern" in rule:
                matches = re.finditer(rule["pattern"], item.text, re.IGNORECASE)
                for match in matches:
                    issues.append(Issue(
                        category=self.category,
                        severity=Severity(rule.get("severity", "warning")),
                        message=rule["message"],

                        rule_id=rule.get("id"),

```typescript
const client = new Anthropic(); // reads ANTHROPIC_API_KEY from env

export async function POST(req: NextRequest) {
  const { question, userId } = await req.json();

  // Step 1: Fetch the user's relevant data
  const userData = await getUserData(userId);

  // Step 2: Stream Claude's response
  const stream = await client.messages.stream({
    model: "claude-sonnet-4-5-20250929",
    max_tokens: 4096,
    system: `You are a helpful assistant that answers questions about the user's data.

Here is the user's data:
${JSON.stringify(userData, null, 2)}`,
    messages: [{ role: "user", content: question }],
  });

  // Step 3: Convert to a ReadableStream for the browser
  const encoder = new TextEncoder();
  const readable = new ReadableStream({
    async start(controller) {
      stream.on("text", (text) => {
        controller.enqueue(encoder.encode(`data: ${JSON.stringify({ text })}\n\n`));
      });

      stream.on("end", () => {
        controller.enqueue(encoder.encode("data: [DONE]\n\n"));
        controller.close();
      });

      stream.on("error", (err) => {
        console.error("Stream error:", err);
        controller.error(err);
      });
    },
  });

  return new Response(readable, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
```

```typescript
// hooks/useAskAI.ts
import { useState, useCallback } from "react";

export function useAskAI() {
  const [answer, setAnswer] = useState("");
  const [isLoading, setIsLoading] = useState(false);

  const ask = useCallback(async (question: string, userId: string) => {
    setAnswer("");
    setIsLoading(true);

    try {
      const res = await fetch("/api/ask", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ question, userId }),
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      const reader = res.body!.getReader();
      const decoder = new TextDecoder();

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        const chunk = decoder.decode(value, { stream: true });
        const lines = chunk.split("\n").filter((l) => l.startsWith("data: "));

        for (const line of lines) {
          const data = line.slice(6); // strip "data: "
          if (data === "[DONE]") break;

          try {
            const { text } = JSON.parse(data);
            setAnswer((prev) => prev + text);
          } catch {
            // skip malformed chunks
          }
        }
      }
    } catch (err) {
      console.error(err);
      setAnswer("Sorry, something went wrong. Please try again.");
    } finally {
      setIsLoading(false);
    }
  }, []);

  return { answer, isLoading, ask };
}
```

```python
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
import time

class PythonFileHandler(FileSystemEventHandler):
    def on_modified(self, event):
        if not event.src_path.endswith(".py"):
            return

        print(f"\n{'='*50}")
        print(f"Change detected: {event.src_path}")
        print(f"{'='*50}")

            (["mypy", event.src_path],           "Mypy Type Check"),

            result = subprocess.run(cmd, capture_output=True, text=True)
            print(result.stdout)
            if result.returncode != 0:
                print(result.stderr)
                print(f"✗ {label} FAILED")
            else:
                print(f"✓ {label} passed")
```

```toml
# .claude/hooks.toml

# Run linting after any file edit
[[hooks.PostToolUse]]
description = "Lint and type-check Python files after editing"
tool = "edit_file"  # triggers after the text editor tool modifies a file
command = """
# Extract the file path from the tool input
FILE="$CLAUDE_TOOL_INPUT_FILE"

# Only run for Python files
if [[ "$FILE" == *.py ]]; then

  ruff check "$FILE" 2>&1

  echo "🔍 Running type checker..."
  mypy "$FILE" 2>&1

  # Exit non-zero if either failed, so Claude sees the error
  if ! ruff check "$FILE" --quiet 2>/dev/null || ! mypy "$FILE" --no-error-summary 2>/dev/null; then
    echo "❌ Linting or type-check issues found above."
    exit 1
  else
    echo "✅ All checks passed."
  fi
fi
"""
```

|---|---|
| **Putting "always lint" in CLAUDE.md** | Relies on the LLM remembering — it can forget, and it costs tokens every attempt |

| **Asking Claude to run checks** | You said you keep forgetting — that's exactly the problem hooks solve |

- **The trigger is deterministic**: "every time a `.py` file is edited" — no judgement needed.
- **The action doesn't need LLM reasoning**: just run `ruff` and `mypy`.
- **Zero token cost**: runs as a shell command outside the model.

# TCP vs UDP

Both are transport-layer protocols that sit on top of IP, but they make fundamentally different tradeoffs.

---

## TCP (Transmission Control Protocol)

**Core idea:** Reliable, ordered, connection-oriented byte stream.

**How it works:**
- **Three-way handshake** establishes a connection before any data flows (SYN → SYN-ACK → ACK).
- Every segment is **acknowledged**; lost segments are **retransmitted**.
- Data arrives **in order** — the receiver reassembles out-of-order packets and holds them until gaps are filled.
- **Flow control** (receiver window) prevents overwhelming a slow receiver.
- **Congestion control** (slow start, AIMD, etc.) adapts sending rate to network conditions.

**Guarantees:** Reliable delivery, ordering, no duplication, error detection.

**Cost:** Higher latency (handshake, retransmissions, head-of-line blocking), more overhead per segment (20-byte minimum header), and state maintained on both ends.

---

## UDP (User Datagram Protocol)

**Core idea:** Unreliable, unordered, connectionless datagrams.

**How it works:**
- **No handshake** — just send a datagram.
- **No acknowledgments**, no retransmission, no ordering guarantees.
- Each datagram is independent; some may arrive out of order, be duplicated, or never arrive at all.
- Minimal header (8 bytes): source port, destination port, length, checksum.

**Guarantees:** Basically just addressing (ports) and an optional integrity checksum. That's it.

**Cost:** Almost none — which is the entire point.

---

## When to Use Each

| Use Case | Protocol | Why |
|---|---|---|
| **Web traffic (HTTP/HTTPS)** | TCP (or QUIC over UDP) | Need reliable, ordered delivery of pages/API responses |
| **File transfer (FTP, SCP)** | TCP | Every byte must arrive correctly |
| **Email (SMTP, IMAP)** | TCP | Messages must be delivered intact |
| **Database connections** | TCP | Queries and results can't have missing pieces |
| **Real-time video/voice (VoIP, video calls)** | UDP | A dropped frame is better than a delayed one; retransmission would arrive too late to be useful |
| **Online gaming (fast-paced)** | UDP | Low latency matters more than perfect reliability; stale state updates are worthless |
| **DNS lookups** | UDP (typically) | Small request/response; avoid handshake overhead; just retry if lost |
| **Streaming media (live)** | UDP (often via RTP) | Smooth playback matters more than perfect fidelity |
| **IoT / sensor telemetry** | UDP | Lightweight; devices are resource-constrained; individual lost readings are tolerable |

---

## The Key Decision Framework

**Choose TCP when:**
- **Correctness matters** — every byte must arrive, in order, exactly once.
- You can tolerate the latency cost of connection setup and retransmission.

**Choose UDP when:**
- **Timeliness matters more than completeness** — late data is as bad as lost data.
- You want to **build your own reliability** on top (many modern protocols do this, e.g., QUIC, custom game protocols that selectively acknowledge only what matters).
- You need **low overhead** — broadcast/multicast, high-frequency small messages.

---

## A Modern Nuance: QUIC

It's worth noting that the line has blurred. **QUIC** (used by HTTP/3) runs over UDP but implements its own reliability, ordering, encryption, and congestion control in userspace. It avoids TCP's head-of-line blocking by multiplexing independent streams — so a lost packet in one stream doesn't stall the others. This is a great example of "choose UDP and build exactly the reliability semantics you need."

## 4. Structural Fix: Avoid Passing Props Altogether

Often the best optimization is **restructuring** so the child doesn't receive changing props:

### Move state closer to where it's used

```jsx
// ❌ Parent state change re-renders everything
function Parent() {
  const [inputValue, setInputValue] = useState('');
  return (
    <div>
      <input value={inputValue} onChange={e => setInputValue(e.target.value)} />
      <ExpensiveList /> {/* Re-renders on every keystroke! */}
    </div>
  );
}

// ✅ Extract the stateful part into its own component
function SearchInput() {
  const [inputValue, setInputValue] = useState('');
  return <input value={inputValue} onChange={e => setInputValue(e.target.value)} />;
}

function Parent() {
  return (
    <div>
      <SearchInput />
      <ExpensiveList /> {/* No longer re-renders */}
    </div>
  );
}
```

### The `children` pattern

```jsx
// ✅ children aren't re-created when Parent's state changes
function AnimatedContainer({ children }) {
  const [pos, setPos] = useState(0);
  useEffect(() => { /* animation updating pos rapidly */ }, []);

  return <div style={{ transform: `translateX(${pos}px)` }}>{children}</div>;
}

// Usage — ExpensiveComponent is created by the grandparent,
// so it's NOT re-rendered when AnimatedContainer's pos changes

  <ExpensiveComponent />
</AnimatedContainer>
```

## Decision Framework

```
Props are only primitives?
  └─ Yes → React.memo on the child. Done.
  └─ No (objects/arrays/functions) →
       Stabilize with useMemo/useCallback in parent
       + React.memo on the child.

Still having issues?
  └─ Can you restructure to avoid passing the prop? (best)
  └─ Do you need deep comparison? → Custom comparator (last resort)
```

**Key principle**: `React.memo` is the shield, but `useMemo`/`useCallback` in the parent are what make the shield effective. Neither works well without the other when non-primitive props are involved.
